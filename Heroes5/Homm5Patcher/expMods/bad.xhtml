function lvl(n) 
    if n==1 return 0
    elseif n==2 return 1000
    elseif n==3 return 2000
    elseif n<10  return 200 + 2*clvl(n-1) - clvl(n-2)
    elseif n<13 return 300 + 2*clvl(n-1) - clvl(n-2)
    elseif n<25 return 1.20 * (clvl(n-1) - clvl(n-2)) + clvl(n-1) 
    else return ((n-1) * 0.04) * (clvl(n-1) - clvl(n-2)) + clvl(n-1)
    end
end

cache = Int128[]

function clvl(n)
    try
        return cache[n]
    catch e
        xp = floor(lvl(n))
        push!(cache, xp)
        return xp
    end
end
